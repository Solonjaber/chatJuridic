# Acesse a pasta do projeto
cd meu-projeto/

# Inicie um repositório git se ainda não tiver
git init
heroku create nome-do-seu-app --buildpack https://github.com/heroku/heroku-buildpack-python

# Adicione o buildpack do APT
heroku buildpacks:add --index 1 https://github.com/heroku/heroku-buildpack-apt

# Confirme a ordem dos buildpacks
heroku buildpacks

# Adicione os arquivos e faça commit
git add .
git commit -m "Deploy inicial"

# Faça o push para o Heroku
git push heroku main

# Variáveis de ambiente
Configure no Heroku Dashboard (ou via CLI)

heroku config:set OPENAI_API_KEY=xxx
heroku config:set OTHER_SECRET=yyy

# Teste final
Acesse o app:
heroku open

Ou veja os logs:
heroku logs --tail
